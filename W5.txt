CREATE VIEW nums2 AS
  SELECT dno,count(ssn) no_emps
  FROM employee
  GROUP BY dno;
CREATE VIEW num3 AS
  SELECT dno
  FROM nums2
  WHERE no_emps > 4;
SELECT dno,count(ssn)
FROM nums2 NATURAL JOIN employee
WHERE salary > 40000
GROUP BY dno;
